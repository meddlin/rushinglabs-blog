Component code

<MyDocHighlighter language="javascriopt">
{`
function Video({ children, src, width }) {
    width = width !== undefined || "" ? width : "100%";
    return (
        <video controls width={width}>
            <source src={src}
                type="video/webm" />
        </video>
    );
}
`}
